# 班次配置（含薪资计算参数）
SHIFT_CONFIG = {
    "白班": {"time_range": "7:00-18:00", "hours": 11.0, "daily_salary": 160.0, "has_customer_service": True},
    "中班": {"time_range": "11:30-00:00", "hours": 12.5, "daily_salary": 190.0, "has_customer_service": False},
    "夜班": {"time_range": "18:00-7:00", "hours": 13.0, "daily_salary": 190.0, "has_customer_service": False}
}

# 价格配置
PRICE_CONFIG = {
    "集包单价": 0.6859,
    "环线单价": 0.276355
}

class PerformanceCalculator:
    @staticmethod
    def calculate_metrics(import_data, personnel_data):
        """
        计算所有绩效指标
        import_data: 导入的原始数据（单条）
        personnel_data: 输入的人数数据
        """
        shift = import_data["班次"]
        shift_info = SHIFT_CONFIG.get(shift, SHIFT_CONFIG["白班"])
        
        # 1. 基础参数提取
        卸车量 = import_data.get("卸车量", 0)
        环线量 = import_data.get("环线量", 0)
        集包量 = import_data.get("集包量", 0)
        
        # 2. 人数提取（默认0）
        卸车人数 = personnel_data.get("卸车人数", 0)
        集包人数 = personnel_data.get("集包人数", 0)
        南环人数 = personnel_data.get("南环人数", 0)
        北环人数 = personnel_data.get("北环人数", 0)
        文件人数 = personnel_data.get("文件人数", 0)
        发验人数 = personnel_data.get("发验人数", 0)
        客服人数 = personnel_data.get("客服人数", 0) if shift_info["has_customer_service"] else 0
        接发员人数 = personnel_data.get("接发员人数", 0)
        
        # 3. 核心计算
        班次单价 = shift_info["daily_salary"]
        班次时长 = shift_info["hours"]
        
        # 卸车相关
        卸车人效 = 卸车量 / 卸车人数 if 卸车人数 > 0 else 0
        卸车薪资 = 卸车人数 * 班次单价 / 班次时长 if 卸车人数 > 0 else 0
        卸车盈亏 = 0 - 卸车薪资
        
        # 集包相关
        集包收入 = 集包量 * PRICE_CONFIG["集包单价"]
        集包薪资 = 集包人数 * 班次单价 / 班次时长 if 集包人数 > 0 else 0
        集包盈亏 = 集包收入 - 集包薪资
        
        # 环线相关
        环线人数 = 南环人数 + 北环人数
        环线收入 = 环线量 * PRICE_CONFIG["环线单价"]
        环线薪资 = 环线人数 * 班次单价 / 班次时长 if 环线人数 > 0 else 0
        环线盈亏 = 环线收入 - 环线薪资
        
        # 其他相关
        其他人数 = 文件人数 + 发验人数 + 客服人数 + 接发员人数
        其他成本 = 其他人数 * 班次单价 / 班次时长 if 其他人数 > 0 else 0
        其他盈亏 = 0 - 其他成本
        
        # 汇总
        总盈亏 = 卸车盈亏 + 集包盈亏 + 环线盈亏 + 其他盈亏
        
        # 格式化结果（保留2位小数）
        return {
            # 原始数据
            "日期": import_data.get("日期", ""),
            "时段": import_data.get("时段", ""),
            "班次": shift,
            "频次": import_data.get("频次", ""),
            "卸车量": 卸车量,
            "环线量": 环线量,
            "集包量": 集包量,
            # 人数数据
            "卸车人数": 卸车人数,
            "集包人数": 集包人数,
            "南环人数": 南环人数,
            "北环人数": 北环人数,
            "环线人数": 环线人数,
            "文件人数": 文件人数,
            "发验人数": 发验人数,
            "客服人数": 客服人数,
            "接发员人数": 接发员人数,
            "其他人数": 其他人数,
            # 计算结果
            "卸车人效": round(卸车人效, 2),
            "卸车薪资": round(卸车薪资, 2),
            "卸车盈亏": round(卸车盈亏, 2),
            "集包收入": round(集包收入, 2),
            "集包薪资": round(集包薪资, 2),
            "集包盈亏": round(集包盈亏, 2),
            "环线收入": round(环线收入, 2),
            "环线薪资": round(环线薪资, 2),
            "环线盈亏": round(环线盈亏, 2),
            "其他成本": round(其他成本, 2),
            "其他盈亏": round(其他盈亏, 2),
            "总盈亏": round(总盈亏, 2)
        }

    @staticmethod
    def parse_excel(file_stream):
        """解析上传的绩效数据模板"""
        try:
            df = pd.read_excel(file_stream, engine="openpyxl")
            # 标准化列名（匹配模板常见列名，兼容不同格式）
            column_mapping = {
                "日期": "日期",
                "时段": "时段",
                "班次": "班次",
                "频次": "频次",
                "卸车量": "卸车量",
                "环线量": "环线量",
                "集包量": "集包量"
            }
            # 筛选有效列，填充缺失值
            valid_columns = [col for col in column_mapping.keys() if col in df.columns]
            df = df[valid_columns].fillna({
                "卸车量": 0,
                "环线量": 0,
                "集包量": 0,
                "频次": "1次"
            })
            # 转换为字典列表
            return df.to_dict("records"), None
        except Exception as e:
            return None, f"Excel解析失败：{str(e)}"

@app.route('/api/import-excel', methods=['POST'])
def import_excel():
    if 'file' not in request.files:
        return jsonify({"success": False, "error": "未选择文件"}), 400
    
    file = request.files['file']
    if file.filename == '':
        return jsonify({"success": False, "error": "文件名不能为空"}), 400
    
    if file and file.filename.endswith(('.xlsx', '.xls')):
        try:
            data_list, error = PerformanceCalculator.parse_excel(file.stream)
            if error:
                return jsonify({"success": False, "error": error}), 400
            return jsonify({
                "success": True,
                "data": data_list,
                "count": len(data_list),
                "message": f"成功导入{len(data_list)}条数据"
            })
        except Exception as e:
            return jsonify({"success": False, "error": f"导入失败：{str(e)}"}), 500
    else:
        return jsonify({"success": False, "error": "仅支持Excel文件（.xlsx/.xls）"}), 400

# 3. 计算绩效指标 API
@app.route('/api/calculate', methods=['POST'])
def calculate_performance():
    try:
        data = request.get_json()
        import_data = data.get("importData", {})  # 单条导入数据
        personnel_data = data.get("personnelData", {})  # 人数输入数据
        
        if not import_data:
            return jsonify({"success": False, "error": "缺少导入数据"}), 400
        
        # 计算指标
        result = PerformanceCalculator.calculate_metrics(import_data, personnel_data)
        return jsonify({
            "success": True,
            "result": result
        })
    except Exception as e:
        return jsonify({"success": False, "error": f"计算失败：{str(e)}"}), 500

# 4. 导出计算结果 API
@app.route('/api/export-result', methods=['POST'])
def export_result():
    try:
        data = request.get_json()
        result_list = data.get("resultList", [])
        
        if not result_list:
            return jsonify({"success": False, "error": "无计算结果可导出"}), 400
        
        # 转换为DataFrame
        df = pd.DataFrame(result_list)
        
        # 生成Excel（带格式）
        output = io.BytesIO()
        with pd.ExcelWriter(output, engine='openpyxl') as writer:
            df.to_excel(writer, sheet_name='绩效计算结果', index=False)
            # 美化格式
            ws = writer.book['绩效计算结果']
            # 设置列宽
            column_widths = {
                'A': 12, 'B': 12, 'C': 8, 'D': 8, 'E': 10, 'F': 10, 'G': 10,
                'H': 10, 'I': 10, 'J': 10, 'K': 10, 'L': 10, 'M': 10, 'N': 10,
                'O': 10, 'P': 10, 'Q': 10, 'R': 10, 'S': 10, 'T': 10, 'U': 10,
                'V': 10, 'W': 10, 'X': 10, 'Y': 10, 'Z': 10, 'AA': 10
            }
            for col, width in column_widths.items():
                if col in ws.column_dimensions:
                    ws.column_dimensions[col].width = width
        
        output.seek(0)
        filename = f"绩效计算结果_{datetime.now().strftime('%Y%m%d%H%M%S')}.xlsx"
        return send_file(
            output,
            mimetype='application/vnd.openxmlformats-officedocument.spreadsheetml.sheet',
            as_attachment=True,
            download_name=filename
        )
    except Exception as e:
        return jsonify({"success": False, "error": f"导出失败：{str(e)}"}), 500

    <script>
        const { createApp } = Vue;
        const { Tab, TabContent } = bootstrap;

        createApp({
            data() {
                return {
                    // 导入相关
                    selectedFile: null,
                    importedData: [],
                    selectedImportDataId: "",
                    selectedImportData: null,
                    loading: false,
                    // 人数输入相关
                    personnelData: {
                        卸车人数: 0,
                        集包人数: 0,
                        南环人数: 0,
                        北环人数: 0,
                        文件人数: 0,
                        发验人数: 0,
                        客服人数: 0,
                        接发员人数: 0
                    },
                    showCustomerService: true,
                    shiftConfig: {},
                    // 计算相关
                    calcLoading: false,
                    currentResult: null,
                    resultList: [],
                    // 导出相关
                    exporting: false
                };
            },
            mounted() {
                // 获取班次配置
                this.getShiftConfig();
                // 初始化标签页
                this.initTabs();
            },
            methods: {
                // 初始化标签页
                initTabs() {
                    document.querySelectorAll('[data-bs-toggle="tab"]').forEach(tab => {
                        new Tab(tab);
                    });
                },
                // 获取班次配置
                async getShiftConfig() {
                    try {
                        const res = await axios.get('/api/shift-config');
                        if (res.data.success) {
                            this.shiftConfig = res.data.config;
                        }
                    } catch (e) {
                        alert('获取班次配置失败：' + e.message);
                    }
                },
                // 选择文件
                handleFileSelect(e) {
                    this.selectedFile = e.target.files[0];
                },
                // 上传Excel
                async uploadExcel() {
                    if (!this.selectedFile) {
                        alert('请先选择Excel文件');
                        return;
                    }

                    this.loading = true;
                    const formData = new FormData();
                    formData.append('file', this.selectedFile);

                    try {
                        const res = await axios.post('/api/import-excel', formData, {
                            headers: { 'Content-Type': 'multipart/form-data' }
                        });

                        if (res.data.success) {
                            this.importedData = res.data.data;
                            this.selectedImportDataId = "";
                            this.selectedImportData = null;
                            alert(res.data.message);
                        } else {
                            alert('导入失败：' + res.data.error);
                        }
                    } catch (e) {
                        alert('上传失败：' + (e.response?.data?.error || e.message));
                    } finally {
                        this.loading = false;
                    }
                },
                // 选择导入的数据
                selectImportData() {
                    if (this.selectedImportDataId === "") {
                        this.selectedImportData = null;
                        this.showCustomerService = true;
                        return;
                    }

                    this.selectedImportData = this.importedData[Number(this.selectedImportDataId)];
                    // 重置人数输入
                    this.personnelData = {
                        卸车人数: 0,
                        集包人数: 0,
                        南环人数: 0,
                        北环人数: 0,
                        文件人数: 0,
                        发验人数: 0,
                        客服人数: 0,
                        接发员人数: 0
                    };
                    // 控制客服人数显示（仅白班）
                    const shift = this.selectedImportData.班次;
                    this.showCustomerService = this.shiftConfig[shift]?.has_customer_service || false;
                },
                // 验证数字输入（仅允许非负整数）
                validateNumber(key) {
                    if (isNaN(this.personnelData[key]) || this.personnelData[key] < 0) {
                        this.personnelData[key] = 0;
                    } else {
                        this.personnelData[key] = Math.floor(this.personnelData[key]);
                    }
                },
                // 计算绩效指标
                async calculatePerformance() {
                    if (!this.selectedImportData) {
                        alert('请先选择一条导入的数据');
                        return;
                    }

                    // 验证至少有一个人数不为0
                    const hasPersonnel = Object.values(this.personnelData).some(v => v > 0);
                    if (!hasPersonnel) {
                        alert('请至少输入一个环节的人数');
                        return;
                    }

                    this.calcLoading = true;
                    try {
                        const res = await axios.post('/api/calculate', {
                            importData: this.selectedImportData,
                            personnelData: this.personnelData
                        });

                        if (res.data.success) {
                            this.currentResult = res.data.result;
                            // 切换到结果标签页
                            document.querySelector('#all-result-tab').click();
                        } else {
                            alert('计算失败：' + res.data.error);
                        }
                    } catch (e) {
                        alert('计算请求失败：' + (e.response?.data?.error || e.message));
                    } finally {
                        this.calcLoading = false;
                    }
                },
                // 添加到历史结果
                addResultToHistory() {
                    if (!this.currentResult) {
                        alert('无计算结果可添加');
                        return;
                    }

                    // 检查是否已存在相同数据
                    const exists = this.resultList.some(item => 
                        item.日期 === this.currentResult.日期 &&
                        item.时段 === this.currentResult.时段 &&
                        item.班次 === this.currentResult.班次
                    );

                    if (exists) {
                        alert('该时段的计算结果已存在');
                        return;
                    }

                    this.resultList.push({ ...this.currentResult });
                    alert('成功添加到历史结果列表');
                },
